Berikut method update yang lebih rapi, aman, dan lengkap.

PHP
public function update(Request $request)
{
    $page = LandingPage::firstOrCreate([]);

    $data = $request->only([
        'hero_title',
        'hero_subtitle',
        'hero_button_text',
        'hero_button_link',

        'app_button_text',
        'app_button_link',

        'about_title',
        'about_description',

        'service_title',
        'service_description',

        'therapist_title',
        'therapist_description',

        'join_title',
        'join_description',

        'download_title',
        'download_description'
    ]);

    /*
    |--------------------------------------------------------------------------
    | HERO IMAGE
    |--------------------------------------------------------------------------
    */

    if ($request->hasFile('hero_image')) {

        $file = $request->file('hero_image');

        $filename = time().'_'.$file->getClientOriginalName();

        $file->move(public_path('images'), $filename);

        $data['hero_image'] = $filename;
    }

    /*
    |--------------------------------------------------------------------------
    | ABOUT IMAGE
    |--------------------------------------------------------------------------
    */

    if ($request->hasFile('about_image')) {

        $file = $request->file('about_image');

        $filename = time().'_'.$file->getClientOriginalName();

        $file->move(public_path('images'), $filename);

        $data['about_image'] = $filename;
    }

    /*
    |--------------------------------------------------------------------------
    | JOIN IMAGE
    |--------------------------------------------------------------------------
    */

    if ($request->hasFile('join_image')) {

        $file = $request->file('join_image');

        $filename = time().'_'.$file->getClientOriginalName();

        $file->move(public_path('images'), $filename);

        $data['join_image'] = $filename;
    }

    /*
    |--------------------------------------------------------------------------
    | DOWNLOAD IMAGE
    |--------------------------------------------------------------------------
    */

    if ($request->hasFile('download_image')) {

        $file = $request->file('download_image');

        $filename = time().'_'.$file->getClientOriginalName();

        $file->move(public_path('images'), $filename);

        $data['download_image'] = $filename;
    }

    /*
    |--------------------------------------------------------------------------
    | UPDATE DATA
    |--------------------------------------------------------------------------
    */

    $page->fill($data);
    $page->save();

    return back()->with('success', 'Landing page updated successfully');
}
